1use bonds_cli::args::{Cli, Commands, ConfigAction};
2mod commands;
3mod ui;
4
5use clap::Parser;
6use commands::{
7    cmd_add, cmd_config_get, cmd_config_set, cmd_info, cmd_list, cmd_migrate, cmd_remove,
8    cmd_update,
9};
10
11fn main() {
12    let cli = Cli::parse();
13
14    let result = match cli.command {
15        Commands::Config { action } => match action {
16            ConfigAction::Get { key } => cmd_config_get(&key),
17            ConfigAction::Set { key, value } => cmd_config_set(&key, &value),
18        },
19        cmd => {
20            // Only init DB for commands that need it.
21            let manager = match bonds_core::BondManager::new(cli.db) {
22                Ok(m) => m,
23                Err(e) => {
24                    // Use the same color path as all other errors.
25                    eprintln!(
26                        "{}",
27                        ui::format_context_error("Failed to initialize bond manager", &e)
28                    );
29                    std::process::exit(1);
30                }
31            };
32
33            match cmd {
34                Commands::Add {
35                    source,
36                    target,
37                    contents,
38                    name,
39                } => cmd_add(&manager, source, target, contents, name),
40                Commands::List => cmd_list(&manager),
41                Commands::Info { id } => cmd_info(&manager, &id),
42                Commands::Remove { id, with_target } => cmd_remove(&manager, &id, with_target),
43                Commands::Config { .. } => unreachable!(),
44                Commands::Update {
45                    id,
46                    source,
47                    target,
48                    name,
49                } => cmd_update(&manager, &id, source, target, name),
50                Commands::Migrate { id, dest } => cmd_migrate(&manager, &id, dest),
51            }
52        }
53    };
54
55    if let Err(e) = result {
56        // One place for every command failure.
57        eprintln!("{}", ui::format_error(&e));
58        std::process::exit(1);
59    }
60}
